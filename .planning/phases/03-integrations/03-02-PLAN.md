import { google } from 'googleapis';
import type { Source, Lead } from '@/generated/prisma/client';

function getSheetsClient() {
  const auth = new google.auth.JWT({
    email: process.env.GOOGLE_SERVICE_ACCOUNT_EMAIL,
    key: process.env.GOOGLE_PRIVATE_KEY?.replace(/\\n/g, '\n'),
    scopes: ['https://www.googleapis.com/auth/spreadsheets'],
  });
  return google.sheets({ version: 'v4', auth });
}

export async function appendLeadToSheet(
  source: Pick<Source, 'sheetsId' | 'sheetTab'>,
  lead: Pick<Lead, 'fields'>
): Promise<void> {
  if (!source.sheetsId || !source.sheetTab) return;

  const fields = lead.fields as Record<string, unknown>;
  // Row order: name, email, phone, utm_source, utm_medium, utm_campaign, utm_term, utm_content, leadscore, leadClass
  const row = [
    fields.name ?? '',
    fields.email ?? '',
    fields.phone ?? '',
    fields.utm_source ?? '',
    fields.utm_medium ?? '',
    fields.utm_campaign ?? '',
    fields.utm_term ?? '',
    fields.utm_content ?? '',
    fields.leadscore ?? '',
    fields.leadClass ?? '',
  ].map(String);

  const sheets = getSheetsClient();
  await sheets.spreadsheets.values.append({
    spreadsheetId: source.sheetsId,
    range: `${source.sheetTab}!A1`,
    valueInputOption: 'USER_ENTERED',
    insertDataOption: 'INSERT_ROWS',
    requestBody: { values: [row] },
  });
}

import type { Source, Lead } from '@/generated/prisma/client';

export async function postToDataCrazy(
  source: Pick<Source, 'dataCrazyUrl' | 'fieldMapping'>,
  lead: Pick<Lead, 'fields'>
): Promise<void> {
  if (!source.dataCrazyUrl) return;

  const leadFields = lead.fields as Record<string, unknown>;
  const mapping = (source.fieldMapping ?? {}) as Record<string, string>;

  // Build body: if mapping has entries, apply it; otherwise send leadFields as-is
  let body: Record<string, unknown>;
  if (Object.keys(mapping).length > 0) {
    body = {};
    for (const [destKey, sourceKey] of Object.entries(mapping)) {
      body[destKey] = leadFields[sourceKey];
    }
  } else {
    body = leadFields;
  }

  const res = await fetch(source.dataCrazyUrl, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });

  if (!res.ok) {
    const text = await res.text().catch(() => '');
    throw new Error(`DataCrazy POST failed: ${res.status} ${text}`);
  }
}

import { prisma } from '@/lib/prisma';

export function computeNextRetryAt(attemptCount: number): Date {
  const delayMs = Math.min(Math.pow(2, attemptCount) * 60_000, 60 * 60_000);
  return new Date(Date.now() + delayMs);
}

export async function updateSyncLogSuccess(id: string): Promise<void> {
  await prisma.syncLog.update({
    where: { id },
    data: { status: 'done', error: null, attemptedAt: new Date() },
  });
}

export async function updateSyncLogFailure(
  id: string,
  error: string,
  attemptCount: number
): Promise<void> {
  await prisma.syncLog.update({
    where: { id },
    data: {
      status: 'failed',
      error,
      attemptCount: attemptCount + 1,
      nextRetryAt: computeNextRetryAt(attemptCount + 1),
      attemptedAt: new Date(),
    },
  });
}

**Step 1 — Create src/lib/integrations/trigger.ts:**
```typescript
import type { Lead, Source, SyncLog } from '@/generated/prisma/client';
import { prisma } from '@/lib/prisma';
import { appendLeadToSheet } from './sheets';
import { postToDataCrazy } from './datacrazy';
import { updateSyncLogSuccess, updateSyncLogFailure } from './retry';

type LeadWithSource = Lead & { source: Source; syncLogs: SyncLog[] };

export async function triggerIntegrations(lead: LeadWithSource): Promise<void> {
  const { source, syncLogs } = lead;

  const sheetsLog = syncLogs.find((l) => l.destination === 'sheets');
  const dataCrazyLog = syncLogs.find((l) => l.destination === 'datacrazy');

  const results = await Promise.allSettled([
    sheetsLog
      ? appendLeadToSheet(source, lead)
          .then(() => updateSyncLogSuccess(sheetsLog.id))
          .catch((err: Error) =>
            updateSyncLogFailure(sheetsLog.id, err.message, sheetsLog.attemptCount)
          )
      : Promise.resolve(),
    dataCrazyLog
      ? postToDataCrazy(source, lead)
          .then(() => updateSyncLogSuccess(dataCrazyLog.id))
          .catch((err: Error) =>
            updateSyncLogFailure(dataCrazyLog.id, err.message, dataCrazyLog.attemptCount)
          )
      : Promise.resolve(),
  ]);

  // Update Lead.status: "synced" only if both destinations succeeded
  const allDone = results.every((r) => r.status === 'fulfilled');
  await prisma.lead.update({
    where: { id: lead.id },
    data: { status: allDone ? 'synced' : 'failed' },
  });
}
```

import { computeNextRetryAt } from './retry';

describe('computeNextRetryAt', () => {
  it('returns ~60s delay for attemptCount=0', () => {
    const before = Date.now();
    const result = computeNextRetryAt(0);
    const after = Date.now();

    expect(diff).toBeLessThanOrEqual(60_100 + (after - before));
